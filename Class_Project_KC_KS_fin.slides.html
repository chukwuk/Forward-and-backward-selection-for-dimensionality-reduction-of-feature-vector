Determination of the features with biggest impact on house prices¶

We used a forward stepwise selection method to find the best feature vectors.

  • this method involves : let us train a linear model by stepwise addition of features:
    y=$_{0}$
    y=$_{0}$+$x_{1}$
    y=$_{0}$+$x_{1}$+$x_{2}$;
    y=$_{0}$+$x_{1}$+$x_{2}$+$x_{3}$;
     ...
    and assess its accuracy by an 80/20 train/test split.

    y=$_{0}$+$x_{1}$;
    y=$_{0}$

function ind_2_choose_b(sub::Array{Float64,1}, num::Int64)
    test = Array{Int64, 1}(undef, (length(sub)-1))
    kk=0;
    for s=1:length(sub)
        if s!=num
            kk+=1
            test[kk] = sub[s]
        end
    end
    return test
end
